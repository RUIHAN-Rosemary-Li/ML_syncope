from PIL import Image
import os
import glob

def combine_png_images(image_paths, image_name, layout=(2, 2), target_size=None):
    images = []
    for path in image_paths:
        img = Image.open(path)
        if target_size:
            img = img.resize(target_size, Image.Resampling.LANCZOS)
        images.append(img)
    
    widths, heights = zip(*(i.size for i in images))
    rows, cols = layout
    total_width = min(widths) * cols
    total_height = min(heights) * rows

    resized_images = []
    for img in images:
            img = img.resize((min(widths), min(heights)), Image.Resampling.LANCZOS)
            resized_images.append(img)
    images = resized_images
    
    combined = Image.new('RGB', (total_width, total_height), color='white')
    for idx, img in enumerate(images):
        row = idx // cols
        col = idx % cols
        x = col * min(widths)
        y = row * min(heights)
        combined.paste(img, (x, y))
    output_path = os.path.join(output_dir, image_name)
    combined.save(output_path, dpi=(300, 300))
    print(f"saved: {output_path}")
    return combined

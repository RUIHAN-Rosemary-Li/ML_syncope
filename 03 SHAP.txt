N_BACKGROUND = 100

def rebuild_dataframe_from_array(x_array, template_df: pd.DataFrame) -> pd.DataFrame:
    x_df = pd.DataFrame(x_array, columns=template_df.columns)

    for col in template_df.columns:
        target_dtype = template_df[col].dtype
        if pd.api.types.is_numeric_dtype(target_dtype):
            x_df[col] = pd.to_numeric(x_df[col], errors="coerce")
        else:
            x_df[col] = x_df[col].astype(str)
    return x_df

def encode_feature_for_plot(series: pd.Series):
    s_num = pd.to_numeric(series, errors="coerce")
    if not s_num.isna().all():
        return s_num.values, None
    codes, uniques = pd.factorize(series.astype(str))
    return codes.astype(float), uniques.tolist()

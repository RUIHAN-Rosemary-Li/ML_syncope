def safe_to_numeric(series: pd.Series) -> pd.Series:
    if pd.api.types.is_numeric_dtype(series):
        return series
    converted = pd.to_numeric(series, errors="coerce")
    if converted.notna().sum() >= max(1, int(series.notna().sum() * 0.8)):
        return converted
    return series


def load_raw_data(file_path: str) -> pd.DataFrame:
    df = pd.read_excel(file_path)
    df.columns = df.columns.str.strip()
    for col in NUMERIC_CANDIDATES:
        if col in df.columns:
            df[col] = safe_to_numeric(df[col])
    return df


def preprocess_for_outcome(
    df_raw: pd.DataFrame,
    outcome_col: str,
    *,
    spo2_strategy: str = "random_normal",
    drop_spo2: bool = False,
) -> pd.DataFrame:
    df = df_raw.copy()

    if outcome_col not in df.columns:
        raise ValueError(f"not found: {outcome_col}")

    df = df.dropna(subset=[outcome_col]).copy()

    for col in OUTCOME_MAP.values():
        if col != outcome_col and col in df.columns:
            df = df.drop(columns=col)

    row_missing_ratio = df.isna().mean(axis=1)
    df = df.loc[row_missing_ratio <= 0.5].copy()

    existing_coag = [c for c in COAG_COLUMNS if c in df.columns]
    if existing_coag:
        df = df.drop(columns=existing_coag)

    if drop_spo2 and SPO2_COL in df.columns:
        df = df.drop(columns=[SPO2_COL])
    elif SPO2_COL in df.columns:
        if spo2_strategy == "random_normal":
            np.random.seed(RANDOM_STATE)
            n_missing = df[SPO2_COL].isna().sum()
            if n_missing > 0:
                values = np.random.uniform(95, 100, size=n_missing)
                df.loc[df[SPO2_COL].isna(), SPO2_COL] = np.round(values, 1)
        elif spo2_strategy == "median":
            median_val = df[SPO2_COL].median(skipna=True)
            if pd.isna(median_val):
                median_val = 98.0
            df[SPO2_COL] = df[SPO2_COL].fillna(median_val)
        elif spo2_strategy == "none":
            pass
        else:
            raise ValueError(f"未知 spo2_strategy: {spo2_strategy}")

    valid_labels = {0, 1}
    unique_labels = set(pd.Series(df[outcome_col]).dropna().unique().tolist())
    if not unique_labels.issubset(valid_labels):
        raise ValueError(f"{outcome_col} not binary: {sorted(unique_labels)}")

    df[outcome_col] = df[outcome_col].astype(int)
    return df.reset_index(drop=True)


def family_from_model_name(model_name: str):
    model_name = str(model_name)
    for family, prefix in MODEL_PREFIX_MAP.items():
        if model_name.startswith(prefix):
            return family
    return None


def get_positive_proba(predictor: TabularPredictor, X: pd.DataFrame, model_name: str) -> np.ndarray:
    proba = predictor.predict_proba(X, model=model_name)
    if isinstance(proba, pd.DataFrame):
        if 1 in proba.columns:
            return proba[1].astype(float).values
        if "1" in proba.columns:
            return proba["1"].astype(float).values
        return proba.iloc[:, -1].astype(float).values
    if isinstance(proba, pd.Series):
        return proba.astype(float).values
    arr = np.asarray(proba)
    if arr.ndim == 2 and arr.shape[1] >= 2:
        return arr[:, -1].astype(float)
    return arr.astype(float)


def calc_metrics(y_true: np.ndarray, y_prob: np.ndarray, threshold: float = 0.5) -> dict:
    y_pred = (y_prob >= threshold).astype(int)
    tn, fp, fn, tp = confusion_matrix(y_true, y_pred, labels=[0, 1]).ravel()
    specificity = tn / max(tn + fp, 1)

    return {
        "threshold": float(threshold),
        "auc": float(roc_auc_score(y_true, y_prob)),
        "auprc": float(average_precision_score(y_true, y_prob)),
        "accuracy": float(accuracy_score(y_true, y_pred)),
        "precision": float(precision_score(y_true, y_pred, zero_division=0)),
        "recall": float(recall_score(y_true, y_pred, zero_division=0)),
        "f1": float(f1_score(y_true, y_pred, zero_division=0)),
        "specificity": float(specificity),
        "brier_score": float(brier_score_loss(y_true, y_prob)),
        "prevalence": float(np.mean(y_true)),
        "tp": int(tp),
        "fp": int(fp),
        "tn": int(tn),
        "fn": int(fn),
    }


def optimize_thresholds(y_true: np.ndarray, y_prob: np.ndarray):
    rows = []
    for thr in THRESHOLD_GRID:
        rows.append(calc_metrics(y_true, y_prob, threshold=float(thr)))
    df = pd.DataFrame(rows)
    best_f1_row = df.sort_values(["f1", "recall", "precision"], ascending=[False, False, False]).iloc[0].to_dict()

    fpr, tpr, thresholds = roc_curve(y_true, y_prob)
    youden_j = tpr - fpr
    idx = int(np.argmax(youden_j))
    youden_threshold = float(thresholds[idx])
    if not (0 <= youden_threshold <= 1):
        youden_threshold = 0.5

    return {
        "default_0.5": calc_metrics(y_true, y_prob, 0.5),
        "f1_best": best_f1_row,
        "youden_best": calc_metrics(y_true, y_prob, youden_threshold),
    }, df


def compute_dca(y_true: np.ndarray, y_prob: np.ndarray, thresholds: np.ndarray = DCA_THRESHOLDS) -> pd.DataFrame:
    n = len(y_true)
    prevalence = np.mean(y_true)
    rows = []
    for pt in thresholds:
        y_pred = (y_prob >= pt).astype(int)
        tn, fp, fn, tp = confusion_matrix(y_true, y_pred, labels=[0, 1]).ravel()
        net_benefit = (tp / n) - (fp / n) * (pt / (1 - pt))
        treat_all = prevalence - (1 - prevalence) * (pt / (1 - pt))
        rows.append({
            "threshold": float(pt),
            "net_benefit_model": float(net_benefit),
            "net_benefit_all": float(treat_all),
            "net_benefit_none": 0.0,
        })
    return pd.DataFrame(rows)


def fit_predictor(
    train_df: pd.DataFrame,
    label_col: str,
    model_path: str,
    hyperparameters: dict = None,
) -> TabularPredictor:
    if os.path.exists(model_path):
        shutil.rmtree(model_path)

    predictor = TabularPredictor(
        label=label_col,
        problem_type="binary",
        eval_metric="roc_auc",
        path=model_path,
    ).fit(
        train_data=TabularDataset(train_df),
        hyperparameters=hyperparameters or NINE_FAMILY_HYPERPARAMETERS,
        presets="best_quality",
        auto_stack=True,
        fit_weighted_ensemble=True,
        verbosity=2,
    )
    return predictor


def save_json(obj, path: str):
    with open(path, "w", encoding="utf-8") as f:
        json.dump(obj, f, ensure_ascii=False, indent=2)


def save_text(text: str, path: str):
    with open(path, "w", encoding="utf-8") as f:
        f.write(text)


def plot_roc_pr_calibration_dca(y_true: np.ndarray, model_prob_map: dict, output_dir: str, outcome_key: str):
    plt.figure(figsize=(8, 6))
    for name, y_prob in model_prob_map.items():
        fpr, tpr, _ = roc_curve(y_true, y_prob)
        auc_val = roc_auc_score(y_true, y_prob)
        plt.plot(fpr, tpr, label=f"{name} (AUC={auc_val:.3f})")
    plt.plot([0, 1], [0, 1], linestyle="--", color="gray")
    plt.xlabel("False Positive Rate")
    plt.ylabel("True Positive Rate")
    plt.title(f"{outcome_key} ROC curves")
    plt.legend(loc="lower right", fontsize=8)
    plt.tight_layout()
    plt.savefig(os.path.join(output_dir, f"{outcome_key}_roc_curves.png"), dpi=200)
    plt.close()

    plt.figure(figsize=(8, 6))
    baseline = np.mean(y_true)
    plt.axhline(baseline, linestyle="--", color="gray", label=f"Baseline prevalence={baseline:.3f}")
    for name, y_prob in model_prob_map.items():
        precision, recall, _ = precision_recall_curve(y_true, y_prob)
        ap = average_precision_score(y_true, y_prob)
        plt.plot(recall, precision, label=f"{name} (AUPRC={ap:.3f})")
    plt.xlabel("Recall")
    plt.ylabel("Precision")
    plt.title(f"{outcome_key} PR curves")
    plt.legend(loc="best", fontsize=8)
    plt.tight_layout()
    plt.savefig(os.path.join(output_dir, f"{outcome_key}_pr_curves.png"), dpi=200)
    plt.close()

    plt.figure(figsize=(8, 6))
    plt.plot([0, 1], [0, 1], linestyle="--", color="gray", label="Ideal")
    for name, y_prob in model_prob_map.items():
        prob_true, prob_pred = calibration_curve(y_true, y_prob, n_bins=N_BINS_CALIBRATION, strategy="quantile")
        brier = brier_score_loss(y_true, y_prob)
        plt.plot(prob_pred, prob_true, marker="o", label=f"{name} (Brier={brier:.3f})")
    plt.xlabel("Predicted probability")
    plt.ylabel("Observed probability")
    plt.title(f"{outcome_key} calibration curves")
    plt.legend(loc="upper left", bbox_to_anchor=(1.02, 1), borderaxespad=0.0, fontsize=8)
    plt.tight_layout()
    plt.savefig(os.path.join(output_dir, f"{outcome_key}_calibration_curves.png"), dpi=200, bbox_inches="tight")
    plt.close()

    plt.figure(figsize=(8, 6))
    for name, y_prob in model_prob_map.items():
        dca_df = compute_dca(y_true, y_prob)
        plt.plot(dca_df["threshold"], dca_df["net_benefit_model"], label=name)
    ref_df = compute_dca(y_true, next(iter(model_prob_map.values())))
    plt.plot(ref_df["threshold"], ref_df["net_benefit_all"], linestyle="--", color="black", label="Treat all")
    plt.plot(ref_df["threshold"], ref_df["net_benefit_none"], linestyle=":", color="black", label="Treat none")
    plt.xlabel("Threshold probability")
    plt.ylabel("Net benefit")
    plt.title(f"{outcome_key} DCA")
    plt.legend(loc="best", fontsize=8)
    plt.tight_layout()
    plt.savefig(os.path.join(output_dir, f"{outcome_key}_dca_curves.png"), dpi=200)
    plt.close()


def plot_best_model_diagnostics(y_true: np.ndarray, y_prob: np.ndarray, outcome_key: str, best_model_name: str, output_dir: str):

    plt.figure(figsize=(8, 6))
    plt.plot(threshold_df["threshold"], threshold_df["accuracy"], label="Accuracy")
    plt.plot(threshold_df["threshold"], threshold_df["precision"], label="Precision")
    plt.plot(threshold_df["threshold"], threshold_df["recall"], label="Recall")
    plt.plot(threshold_df["threshold"], threshold_df["f1"], label="F1")
    best_f1_threshold = threshold_df.loc[threshold_df["f1"].idxmax(), "threshold"]
    plt.axvline(best_f1_threshold, linestyle="--", label=f"Best F1={best_f1_threshold:.2f}")
    plt.xlabel("Threshold")
    plt.ylabel("Metric")
    plt.title(f"{outcome_key} threshold optimization - {best_model_name}")
    plt.legend()
    plt.tight_layout()
    plt.savefig(os.path.join(output_dir, f"{outcome_key}_best_model_thresholds.png"), dpi=200)

    cm = confusion_matrix(y_true, y_pred)
    plt.figure(figsize=(5, 5))
    plt.imshow(cm, interpolation="nearest")
    plt.title(f"{outcome_key} confusion matrix - {best_model_name}")
    plt.colorbar()

    plt.xlabel("Predicted label")
    plt.tight_layout()
    plt.savefig(os.path.join(output_dir, f"{outcome_key}_best_model_confusion_matrix.png"), dpi=200)

def top_numeric_features_for_pdp(df: pd.DataFrame, feature_importance_df: pd.DataFrame, max_features: int = 3):
    numeric_cols = set(df.select_dtypes(include=[np.number]).columns.tolist())
    candidates = []
    for feat in feature_importance_df["feature"].tolist():
        if feat in numeric_cols:
            candidates.append(feat)
        if len(candidates) >= max_features:
            break
    return candidates


def generate_partial_dependence_plots(
    predictor: TabularPredictor,
    X: pd.DataFrame,
    feature_importance_df: pd.DataFrame,
    best_model_name: str,
    output_dir: str,
    outcome_key: str,
):
    saved = []
    for feat in top_numeric_features_for_pdp(X, feature_importance_df):
        values = X[feat].dropna()
        if values.nunique() < 5:
            continue
        grid = np.quantile(values, np.linspace(0.05, 0.95, 20))
        pdp_rows = []
        X_tmp = X.copy()
        for val in grid:
            X_tmp[feat] = val
            y_prob = get_positive_proba(predictor, X_tmp, best_model_name)
            pdp_rows.append({"feature_value": float(val), "mean_predicted_risk": float(np.mean(y_prob))})
        pdp_df = pd.DataFrame(pdp_rows)
        plt.figure(figsize=(6, 4))
        plt.plot(pdp_df["feature_value"], pdp_df["mean_predicted_risk"], marker="o")
        plt.xlabel(feat)
        plt.ylabel("Mean predicted risk")
        plt.title(f"{outcome_key} PDP - {feat}")
        plt.tight_layout()
        out_path = os.path.join(output_dir, f"{outcome_key}_pdp_{feat}.png")
        plt.savefig(out_path, dpi=200)
        plt.close()
        pdp_df.to_csv(os.path.join(output_dir, f"{outcome_key}_pdp_{feat}.csv"), index=False)
        saved.append(out_path)
    return saved

def summarize_best_models(leaderboard_df: pd.DataFrame):
    lb = leaderboard_df.copy()
    lb["base_family"] = lb["model"].apply(family_from_model_name)
    best_per_family = (
        lb[lb["base_family"].notna()]
        .sort_values("score_test", ascending=False)
        .groupby("base_family", as_index=False)
        .first()
        .sort_values("score_test", ascending=False)
    )
    weighted_ensemble_rows = lb[lb["model"].astype(str).str.startswith("WeightedEnsemble")].copy()
    return best_per_family, weighted_ensemble_rows

def outcome_analysis(df_raw: pd.DataFrame, outcome_key: str, base_output_dir: str) -> dict:
    outcome_col = OUTCOME_MAP[outcome_key]
    outcome_dir = os.path.join(base_output_dir, outcome_key)
    plots_dir = os.path.join(outcome_dir, "plots")
    os.makedirs(plots_dir, exist_ok=True)

    df_processed = preprocess_for_outcome(df_raw, outcome_col, spo2_strategy="random_normal", drop_spo2=False)
    train_df, test_df = train_test_split(
        df_processed,
        test_size=TEST_SIZE,
        random_state=RANDOM_STATE,
        stratify=df_processed[outcome_col],
    )

    predictor = fit_predictor(train_df, outcome_col, os.path.join(outcome_dir, "ag_model"))
    test_data = TabularDataset(test_df)

    performance = predictor.evaluate(test_data)
    leaderboard = predictor.leaderboard(test_data, silent=True).sort_values("score_test", ascending=False)
    best_per_family, weighted_ensembles = summarize_best_models(leaderboard)

    leaderboard.to_csv(os.path.join(outcome_dir, f"{outcome_key}_leaderboard_full.csv"), index=False)
    best_per_family.to_csv(os.path.join(outcome_dir, f"{outcome_key}_best_per_family.csv"), index=False)
    weighted_ensembles.to_csv(os.path.join(outcome_dir, f"{outcome_key}_weighted_ensemble_rows.csv"), index=False)
    save_json({k: float(v) for k, v in performance.items()}, os.path.join(outcome_dir, f"{outcome_key}_performance.json"))
    save_text(str(predictor.model_best), os.path.join(outcome_dir, f"{outcome_key}_model_best.txt"))

    X_test = test_df.drop(columns=[outcome_col]).copy()
    y_test = test_df[outcome_col].values

    model_eval_list = []
    for _, row in best_per_family.iterrows():
        model_eval_list.append({"group": row["base_family"], "model": row["model"]})
    for _, row in weighted_ensembles.iterrows():
        model_eval_list.append({"group": row["model"], "model": row["model"]})

    model_prob_map = {}
    model_metrics_rows = []
    calibration_detail_list = []
    dca_detail_list = []
    roc_detail_rows = []
    pr_detail_rows = []

    for item in model_eval_list:
        group = item["group"]
        model_name = item["model"]
        y_prob = get_positive_proba(predictor, X_test, model_name)
        model_prob_map[group] = y_prob

        row = calc_metrics(y_test, y_prob, 0.5)
        row.update({"group": group, "model": model_name})
        model_metrics_rows.append(row)

        prob_true, prob_pred = calibration_curve(y_test, y_prob, n_bins=N_BINS_CALIBRATION, strategy="quantile")
        calibration_detail_list.append(pd.DataFrame({
            "group": group,
            "model": model_name,
            "mean_predicted_probability": prob_pred,
            "observed_event_rate": prob_true,
        }))

        dca_df = compute_dca(y_test, y_prob)
        dca_df["group"] = group
        dca_df["model"] = model_name
        dca_detail_list.append(dca_df)

        fpr, tpr, roc_thresholds = roc_curve(y_test, y_prob)
        roc_detail_rows.extend(
            {"group": group, "model": model_name, "fpr": float(a), "tpr": float(b), "threshold": float(c if np.isfinite(c) else 1.0)}
            for a, b, c in zip(fpr, tpr, roc_thresholds)
        )

        precision, recall, pr_thresholds = precision_recall_curve(y_test, y_prob)
        thresholds_for_rows = np.concatenate([[1.0], pr_thresholds])
        pr_detail_rows.extend(
            {"group": group, "model": model_name, "precision": float(a), "recall": float(b), "threshold": float(c)}
            for a, b, c in zip(precision, recall, thresholds_for_rows)
        )

    model_metrics_df = pd.DataFrame(model_metrics_rows).sort_values(["auc", "auprc", "f1"], ascending=False)
    model_metrics_df.to_csv(os.path.join(outcome_dir, f"{outcome_key}_model_metrics_default_threshold.csv"), index=False)
    pd.concat(calibration_detail_list, ignore_index=True).to_csv(os.path.join(outcome_dir, f"{outcome_key}_calibration_detail.csv"), index=False)
    pd.concat(dca_detail_list, ignore_index=True).to_csv(os.path.join(outcome_dir, f"{outcome_key}_dca_detail.csv"), index=False)
    pd.DataFrame(roc_detail_rows).to_csv(os.path.join(outcome_dir, f"{outcome_key}_roc_detail.csv"), index=False)
    pd.DataFrame(pr_detail_rows).to_csv(os.path.join(outcome_dir, f"{outcome_key}_pr_detail.csv"), index=False)

    plot_roc_pr_calibration_dca(y_test, model_prob_map, plots_dir, outcome_key)

    best_overall_group = model_metrics_df.iloc[0]["group"]
    best_overall_model = model_metrics_df.iloc[0]["model"]
    best_overall_prob = model_prob_map[best_overall_group]
    threshold_summary, threshold_detail = optimize_thresholds(y_test, best_overall_prob)

    threshold_rows = []
    for method, values in threshold_summary.items():
        row = dict(values)
        row["method"] = method
        row["group"] = best_overall_group
        row["model"] = best_overall_model
        threshold_rows.append(row)
    pd.DataFrame(threshold_rows).to_csv(os.path.join(outcome_dir, f"{outcome_key}_threshold_summary_best_model.csv"), index=False)
    threshold_detail.to_csv(os.path.join(outcome_dir, f"{outcome_key}_threshold_detail_best_model.csv"), index=False)

    plot_best_model_diagnostics(y_test, best_overall_prob, outcome_key, best_overall_model, plots_dir)

    fi = predictor.feature_importance(test_data, model=best_overall_model)
    fi = fi.reset_index().rename(columns={"index": "feature"})
    fi.to_csv(os.path.join(outcome_dir, f"{outcome_key}_feature_importance_best_model.csv"), index=False)

    generate_partial_dependence_plots(predictor, X_test, fi, best_overall_model, plots_dir, outcome_key)
    optional_shap_analysis(outcome_dir, outcome_key)

    best_base_family = best_per_family.sort_values("score_test", ascending=False).iloc[0]["base_family"]

    summary = {
        "outcome_key": outcome_key,
        "outcome_column": outcome_col,
        "n_total": int(len(df_processed)),
        "positive_rate": float(df_processed[outcome_col].mean()),
        "global_best_group": str(best_overall_group),
        "global_best_model": str(best_overall_model),
        "global_best_auc": float(model_metrics_df.iloc[0]["auc"]),
        "global_best_auprc": float(model_metrics_df.iloc[0]["auprc"]),
        "global_best_recall": float(model_metrics_df.iloc[0]["recall"]),
        "global_best_precision": float(model_metrics_df.iloc[0]["precision"]),
        "global_best_f1": float(model_metrics_df.iloc[0]["f1"]),
        "global_best_brier": float(model_metrics_df.iloc[0]["brier_score"]),
        "best_base_family_for_stability": str(best_base_family),
       }
    save_json(summary, os.path.join(outcome_dir, f"{outcome_key}_summary.json"))
    return summary

def main():
    setup_matplotlib_font()

    parser = argparse.ArgumentParser(description="Three-outcome AutoGluon overnight pipeline")
    parser.add_argument("--file", type=str, default="C:/Users/Rosemary/ag_project/data0.xlsx")
    parser.add_argument("--outdir", type=str, default="C:/Users/Rosemary/ag_project/0416")
    args = parser.parse_args()

    timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
    input_dir = str(Path(args.file).resolve().parent)
    output_dir = args.outdir or os.path.join(input_dir, f"autogluon_three_outcomes_results_{timestamp}")
    os.makedirs(output_dir, exist_ok=True)

    error_logs = []
    summaries = []

    try:
        df_raw = load_raw_data(args.file)
        save_text("\n".join(df_raw.columns.tolist()), os.path.join(output_dir, "raw_columns.txt"))

        for outcome_key in ["1w", "1m", "1y"]:
            try:
                print(f"===== START {outcome_key} =====", flush=True)
                summary = outcome_analysis(df_raw, outcome_key, output_dir)
                summaries.append(summary)
                print(f"===== END {outcome_key} =====", flush=True)
            except Exception as e:
                msg = f"[{outcome_key}] FAILED: {e}\n{traceback.format_exc()}"
                print(msg, flush=True)
                error_logs.append(msg)

        if summaries:
            pd.DataFrame(summaries).to_csv(os.path.join(output_dir, "all_outcomes_summary.csv"), index=False)

        build_summary_notebook(output_dir, os.path.join(output_dir, "view_all_outcomes_results.ipynb"))

    finally:
        if error_logs:
            save_text("\n\n".join(error_logs), os.path.join(output_dir, "errors.log"))
        print(f"All done. Results saved to: {output_dir}", flush=True)


if __name__ == "__main__":
    main()
